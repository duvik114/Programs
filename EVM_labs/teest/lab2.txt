void method(long k,long n,long mas[])
{
    for(long i=n-1;i>=0;i--)
    {
        if(pow(2,i)<=k)
        {
            mas[n-i-1]=1;
            k-=pow(2,i);
        }
        else
        {
            mas[n-i-1]=0;
        }
    }
}
